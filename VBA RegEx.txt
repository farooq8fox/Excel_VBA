Option Explicit

' This user defined function can be used on sheet to get regex matches from cell strings

Sub REGEX_GET()

  ' Clear any existing results
  Sheet2.Range("B2:B1618").ClearContents
  
  ' Get the range
  Dim rg As Range
  Set rg = Sheet2.Range("A2:A1618")
  
  ' Create the regular expression
  Dim i As Long, regex As New RegExp
  regex.Global = True
  regex.Pattern = "[A-Z0-9]{3}\-\d{5}"
  
  
  'Read through the filenames
   For i = 1 To rg.Rows.Count
        ' Use Test to see if the pattern exists
        rg.Cells(i, 2).Value = regex.Test(rg.Cells(i, 1).Value)
    Next i

End Sub
